\section{\makebox[\widthof{\faUsers}][c]{\color{CVBlue}\faUsers}\ 项目经历}
\textbf{基于末端力控的机械臂柔顺轨迹跟踪系统} \hfill 2024.05 -- 2025.05 \titlebreak
项目描述：在六自由度机械臂平台上实现柔顺控制，并能够对未知曲面进行跟踪；在此基础上，基于触觉信息使用 LSTM 架构实现夹持柔性工具的轨迹跟踪。
\begin{itemize}[nosep, topsep=\intraProjectListTopSep]
    \item \textbf{柔顺控制：}基于 ROS 部署传感器与机械臂平台通信，在商家代码基础上完善 ROS 框架下的机械臂位置与速度控制，编写导纳控制算法并在实机上完成验证。
    \item \textbf{数据与模型：}设计数据收集实验方法，在力位控制条件下使用触觉传感器采集夹爪夹持柔性工具的触觉信息，作为模型的时序特征输入；选取 ACTPP 作为时序模型框架构建触觉信息与环境接触特征之间的关系，并利用预测的接触力信息实现跟踪柔顺控制。
\end{itemize}


\textbf{基于人类视频的具身操作策略} \hfill 2026.01 -- 至今 \titlebreak
项目描述：通过从人类视频中提取人手关键点计算位姿，获得人类动作，并经过diffusion policy去噪弥合具身差异，用以微调VLA
\begin{itemize}[nosep, topsep=\intraProjectListTopSep]
    \item \textbf{人手关键点提取：}使用hamer提取人手关键点，Depth AnythingV2获取深度图，Grounded SAM分割目标物体掩码，模拟机械臂动作。
    \item \textbf{diffusion policy：}基于官方源码修改进行训练，支持从带噪样本去噪恢复
    \item \textbf{VLA：}进行VLA微调，使用从人类视频中提取的动作进行微调训练，在libero测试
\end{itemize}


\textbf{VLM-VLABenchmark} \hfill 2026.05 -- 至今 \titlebreak
项目描述：使用机器人操作数据构建VQA，用以测试VLM在具身操作任务中的表现，并研究对以其为骨干的VLA实际表现的影响。
\begin{itemize}[nosep, topsep=\intraProjectListTopSep]
    \item \textbf{任务构建：}对单视角、多视角、历史序列输入数据进行VQA任务构建，任务包含高层次任务规划、低层次轨迹和抓取位姿决定、空间和环境感知等10个任务
    \item \textbf{数据筛选：}从Droid、AgibotWorld、Robomind、Robotwins挖掘机器人操作数据，使用亮度、对比度、模糊程度等指标对视频进行抽帧初筛，使用Qwen3.5-35B-A3B判别是否适合对应任务，进行复筛
    \item \textbf{QA生成：}部分问题利用原数据信息设置答案和干扰项；空间感知类任务使用VLM找物体+Grounded SAM进行标注；抓取类任务使用Grasp Gen进行位姿标注
    \item \textbf{测试：}在闭源高级模型和人类测试验证数据可靠性；对VLA任务常见较小参数开源模型（Qwen系列等）进行VLA和VLM同步测试，进行分任务、输入类型的数据相关性分析
\end{itemize}
